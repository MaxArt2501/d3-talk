Line charts

 
import { line } from 'd3-shape';
const data = [
  [ 19, 19, 24, 13, 20, 24, 14, 57, 16, 60, 30 ],
  [ 13, 36, 14, 24, 31, 18, 26, 40, 10, 29 ],
  [ 43, 18, 15, 30, 21, 19, 65, 20, 18, 14 ]
];
const dataLine = line()
  .x((_, i) => scaleX(i))
  .y(d => scaleY(d));
const lines = select('g.chart')
  .selectAll('path').data(data);
lines.enter()
  .append('path')
  .attr('d', dataLine);
